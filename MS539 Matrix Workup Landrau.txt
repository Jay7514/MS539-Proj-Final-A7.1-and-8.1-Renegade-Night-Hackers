        }

        private void Update(object sender, EventArgs e)
        {
            if (Core.IsUp)
                Top -= speed;
            if (Core.IsDown) 
                Top += speed;
            if (Core.IsLeft)
                Left -= speed;
            if (Core.IsRight)
                Left += speed;

                
        }
